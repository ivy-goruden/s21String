#include <check.h>
#include <string.h>

#include "../s21_string.h"

// Тест 1: Поиск существующего символа
START_TEST(test_find_existing_char) {
  char data[] = "Hello, world!";
  int ch = 'w';
  size_t n = strlen(data);

  char *result = s21_memchr(data, ch, n);
  char *expected = memchr(data, ch, n);

  ck_assert_ptr_eq(result, expected);
  ck_assert_str_eq(result, result);
}
END_TEST

// Создание тестового набора
Suite *memchr_suite(void) {
  Suite *s = suite_create("memchr_tests");

  TCase *tc_core = tcase_create("Core");
  tcase_add_test(tc_core, test_find_existing_char);
  suite_add_tcase(s, tc_core);
  return s;
}

// Запуск тестов
int main_memchr(void) {
  int number_failed;
  Suite *s;
  SRunner *sr;

  s = memchr_suite();
  sr = srunner_create(s);

  srunner_run_all(sr, CK_VERBOSE);
  number_failed = srunner_ntests_failed(sr);
  srunner_free(sr);

  return (number_failed == 0) ? EXIT_SUCCESS : EXIT_FAILURE;
}
